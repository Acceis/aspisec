def initialize(tool_conf, feature_name)
  @name = feature_name
  @path = Pathname.new(tool_conf.dig('location', @name, 'path'))
  @enabled = tool_conf.dig('location', @name).fetch('enabled', true)
  @description = tool_conf.dig('location', @name).fetch('description', '')
end
